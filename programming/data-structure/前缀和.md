
## 1. 前缀和
https://leetcode.cn/problems/range-sum-query-immutable/

```
class NumArray {
public:
    vector<int> res;
    NumArray(vector<int>& nums) {
        //   -2 0 3 -5 2 -1
        // 0 -2 -2 1 -4 -2 -3
        res.resize(nums.size()+1);
        for(int i=0; i < nums.size(); i++) {
            res[i+1] = res[i] + nums[i];
        }
    }
    
    int sumRange(int left, int right) {
        return res[right+1] - res[left];
    }
};
```


## 2.前缀和变种
https://leetcode.cn/problems/count-vowel-strings-in-ranges/description/
```
    vector<int> vowelStrings(vector<string>& words, vector<vector<int>>& queries) {
        vector<int> res;

        unordered_set <char> s = {'a', 'e', 'i', 'o', 'u'};
        vector<int> nums(words.size()+1, 0);
        //    "aba","bcb","ece","aa","e"
        //0   1       1     2     3   4
        for(int i=0; i < words.size(); i++) {
            string t = words[i];
            if(s.count(t[0]) && s.count(t[t.size()-1])) {
                //cout << t << endl;
                nums[i+1] = nums[i] + 1;
            } else{
                nums[i+1] = nums[i];
            }
        }
        for(auto q: queries) {
            int t = nums[q[1] + 1] - nums[q[0]];
            res.push_back(t);
        }
        return res;
    }
```
